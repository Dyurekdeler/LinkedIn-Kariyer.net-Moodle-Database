SELECT COUNT(member_id),OR.name,  OF.name , county, city, street
FROM tbl_works_for W
INNER JOIN tbl_office OF ON W.org_id = OF.org_id
INNER JOIN tbl_organization 
INNER JOIN tbl_address A ON OF.address_id = A.address_id
GROUP BY member_id
HAVING COUNT(member_id)>3
